// 2-2 Q1. Write a program to calculate the sum of even 
// positive numbers up to 30.
// The answer should be 240.

let sum = 0;
for (let k = 1; k <= 10; k++) {
  sum += k;
}
console.log(`The sum of 1+2+...+10 = ${sum}.`);
